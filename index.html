package main

import (
    "log"
    "net/http"
    "html/template"
    "path/to/handler"
    "path/to/middleware"
)

type Templates struct {
    templates *template.Template
}

func main() {
    router := http.NewServeMux()

    router.Handle("/web/static/", 
        http.StripPrefix("/web/static/", 
        http.FileServer(http.Dir("web/static"))))

    router.HandleFunc("/", handler.HomePage)

    log.Fatal(http.ListenAndServe(":3000", middleware.Logging(router)))
}
